! block by keyword
! (this will get any tag with 'Angst' or 'angst' in it)
! (so it will also target tags like 'No Angst')
archiveofourown.org#?#li.group:-abp-has(li:-abp-contains(/Angst|angst/))


! block by exact tag
! (this will target the exact tag 'Angst' or 'angst')
! (it's more strict, which means it will miss tags 'Heavy Angst')
archiveofourown.org#?#li.group:-abp-has(a[href*="/Angst/" i])
